=== 最終 SQL ===
```sql
/* 竹北分公司 ─ 各營業員近 12 個月份
   「月均交易金額 > 500,000 元」之客戶(帳號)數                           */

WITH /* 1. 最近 12 個月份區間(含當月) */
max_mon AS (
    SELECT MAX(SNAP_YYYYMM) AS max_yyyymm
    FROM   DM_S_VIEW.M_AC_ACCOUNT_PROD_TXN        -- 交易歸戶檔
),
period AS (
    SELECT  max_yyyymm,
            TO_CHAR( ADD_MONTHS(TO_DATE(max_yyyymm||'01','YYYYMMDD'),-11)
                   ,'YYYYMM')       AS min_yyyymm
    FROM    max_mon
),

/* 2. 竹北分公司 ─ 有效帳戶(取帳戶主檔最新快照) */
acct_base AS (
    SELECT  ACCT_NBR ,
            SALES_NBR ,
            SALES_NAME
    FROM    DM_S_VIEW.M_AC_ACCOUNT
    WHERE   SNAP_DATE        = (SELECT MAX(SNAP_DATE)
                                FROM DM_S_VIEW.M_AC_ACCOUNT)
      AND   ACCT_VALID_FLAG  = 'Y'
      AND   BRANCH_NAME      = '竹北分公司'
),

/* 3. 帳號 12 個月總交易金額 */
acct_12m_txn AS (
    SELECT  a.SALES_NBR ,
            a.SALES_NAME ,
            t.ACCT_NBR ,
            SUM(t.TXN_AMT_TWD) AS total_txn_amt
    FROM        acct_base           a
    JOIN        DM_S_VIEW.M_AC_ACCOUNT_PROD_TXN t
           ON   t.ACCT_NBR = a.ACCT_NBR
    CROSS JOIN  period              p
    WHERE       t.SNAP_YYYYMM BETWEEN p.min_yyyymm AND p.max_yyyymm
    GROUP BY    a.SALES_NBR ,
                a.SALES_NAME ,
                t.ACCT_NBR
),

/* 4. 換算月均並先行篩選 > 50 萬 */
acct_avg_txn AS (
    SELECT  SALES_NBR ,
            SALES_NAME ,
            ACCT_NBR ,
            total_txn_amt / 12 AS avg_month_txn_amt
    FROM    acct_12m_txn
    WHERE   total_txn_amt / 12 > 500000          -- 月均 > 50 萬
)

/* 5. 依營業員統計客戶(帳號)數 */
SELECT  SALES_NBR      AS 營業員代碼 ,
        SALES_NAME     AS 營業員姓名 ,
        COUNT(DISTINCT ACCT_NBR) AS 月均交易量大於50萬之客戶數
FROM    acct_avg_txn
GROUP   BY SALES_NBR , SALES_NAME
ORDER   BY 月均交易量大於50萬之客戶數 DESC ,
         SALES_NBR;
```
